def add(num1,num2):
 return num1+num2
def sub(num1,num2):
 return num1-num2
def mul(num1,num2):
 return num1*num2
def div(num1,num2):
 return num1/num2
def mod(num1,num2):
 return num1%num2
num1=int(input("Enter number 1: "))
num2=int(input("Enter number 2: "))
print("Addition = ",add(num1,num2))
print("Subtraction = ",sub(num1,num2))
print("Multiplication = ",mul(num1,num2))
print("Division = ",div(num1,num2))
print("Modular = ",mod(num1,num2))
